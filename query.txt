---------------------------------
Daftar Penduduk (JOIN 10 Tabel)
---------------------------------
SELECT 
	penduduk.nik,
    penduduk.nomor_kk,
    penduduk.nama,
    dt_jenis_kelamin.nama jenis_kelamin,
    penduduk.tanggal_lahir tanggal_lahir,
    penduduk.tempat_lahir tempat_lahir,
    dt_hubungan_dalam_keluarga.nama hubungan_dalam_keluarga,
    dt_agama.nama agama,
    dt_pendidikan_terakhir.nama pendidikan_terakhir,
    dt_pekerjaan.nama pekerjaan,
    dt_status_perkawinan.nama status_perkawinan,
    dt_status_penduduk.nama status_penduduk,
    penduduk.nik_ayah,
    penduduk.nama_ayah,
    penduduk.nik_ibu,
    penduduk.nama_ibu,
    rt.nomor rt,
    rw.nomor rw,
    dusun.nama dusun,
    penduduk.tanggal_rekam
FROM penduduk
INNER JOIN dt_jenis_kelamin
	ON dt_jenis_kelamin.id = penduduk.id_jenis_kelamin
INNER JOIN dt_hubungan_dalam_keluarga
	ON dt_hubungan_dalam_keluarga.id = penduduk.id_hubungan_dalam_keluarga
INNER JOIN dt_agama
	ON dt_agama.id = penduduk.id_agama
INNER JOIN dt_pendidikan_terakhir
	ON dt_pendidikan_terakhir.id = penduduk.id_pendidikan_terakhir
INNER JOIN dt_pekerjaan
	ON dt_pekerjaan.id = penduduk.id_pekerjaan
INNER JOIN dt_status_perkawinan
	ON dt_status_perkawinan.id = penduduk.id_status_perkawinan
INNER JOIN dt_status_penduduk
	ON dt_status_penduduk.id = penduduk.id_status_penduduk
INNER JOIN rt
	ON rt.id = penduduk.id_rt
INNER JOIN rw
	ON rw.id = rt.id_rw
INNER JOIN dusun
	ON dusun.id = rw.id_dusun
---------------------------------
Jumlah Records di 4 Tabel untuk dashboard 
---------------------------------
SELECT 
	(SELECT COUNT(*) FROM dusun) AS dusun,
    (SELECT COUNT(*) FROM penduduk) AS penduduk,
	(SELECT COUNT(DISTINCT nomor_kk) FROM penduduk) AS keluarga,
	(SELECT COUNT(*) FROM perangkat_desa) AS perangkat_desa	
---------------------------------
Daftar Kepala Dusun (JOIN 2 Tabel)
---------------------------------
SELECT
	dusun.nik_kepala_dusun as nik_kepala_dusun,
	penduduk.nama as nama_kepala_dusun,
	dusun.nama as nama_dusun
FROM dusun
INNER JOIN penduduk
	ON penduduk.nik = dusun.nik_kepala_dusun
---------------------------------
Daftar Kepala RW (JOIN 3 Tabel)
---------------------------------
SELECT
	rw.nik_kepala_rw as nik_kepala_rw,
	penduduk.nama as nama_kepala_rw,
	rw.nomor as rw,
	dusun.nama as dusun
FROM rw 
INNER JOIN dusun
	ON dusun.id = rw.id_dusun
INNER JOIN penduduk
	ON penduduk.nik = rw.nik_kepala_rw
---------------------------------
Daftar Kepala RT (JOIN 4 Tabel)
---------------------------------
SELECT
	rt.nik_kepala_rt as nik_kepala_rt,
	penduduk.nama as nama_kepala_dusun,
	rt.nomor as rt,
	rw.nomor as rw,
	dusun.nama as dusun
FROM rt
INNER JOIN penduduk
	ON penduduk.nik = rt.nik_kepala_rt
INNER JOIN rw
	ON rw.id = rt.id_rw
INNER JOIN dusun
	ON dusun.id = rw.id_dusun
